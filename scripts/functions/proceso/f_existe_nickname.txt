funciÃ³n que recibe un parametro, el nickname, y retorna true si existe en la tabla jugadores, o false caso contrario.

create or replace function f_existe_nickname (nombre VARCHAR2)
return number as
	res number;
begin
	declare
		vnombre jugadores.nickname%type := '';
	begin
		select
			nickname
		into vnombre
		from jugadores
		where nickname = nombre;
		if length(vnombre) > 0 then
			res := 1;
		end if;
	exception
		when NO_DATA_FOUND then
			-- coordenada invalida
			res := -1;
		when VALUE_ERROR then
			-- Excede el numero de caracteres permitidos
			res := -2;
		when OTHERS then
			res := -3;
	end;
	return res;
end;
